      label: "cloudamqp-dev-n/a",
      plan: "lemur",
      credentials: {
        uri: "amqp://vzpfjola:fGfyADThftxx0EGbHxLEGdIj3aLs5rQW@lemur.cloudamqp.com/vzpfjola"
      }
    }
  ]
}
~~~
