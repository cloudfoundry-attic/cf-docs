{
  elephantsql-dev-n/a: [
  {
    name: "elephantsql-dev-c6c60",
    label: "elephantsql-dev-n/a",
    plan: "turtle",
    credentials: {
      uri: "postgres://seilbmbd:PHxTPJSbkcDakfK4cYwXHiIX9Q8p5Bxn@babar.elephantsql.com:5432/seilbmbd"
    }
  }
  ]
}
~~~
